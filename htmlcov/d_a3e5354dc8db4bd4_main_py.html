140def start_application() -> FastAPI: 

141 """ 

142 Create and initialise the application 

143 

144 It performs the following initialisation steps: 

145 

146 1. Collect all available routers 

147 2. Set-up CORS middleware 

148 

149 Returns: 

150 app: 

151 The fully initialised application. 

152 """ 

153 app = FastAPI() 

154 collect_routers(app) 

155 settings = Inject(get_settings) 

156 app.add_middleware( 

157 CORSMiddleware, 

158 allow_origins=settings.FRACTAL_CORS_ALLOW_ORIGIN.split(";"), 

159 allow_methods=["GET", "POST", "PUT", "PATCH", "DELETE", "OPTIONS"], 

160 allow_headers=[ 

161 "set-cookie", 

162 "Set-Cookie", 

163 "Content-Type", 

164 "Access-Control-Allow-Headers", 

165 "X-Requested-With", 

166 ], 

167 allow_credentials=True, 

168 ) 

169 

170 return app
