76 Args:

85 def __init__(

95 async def get(self, id: ID) -> Optional[UP]:

99 async def get_by_email(self, email: str) -> Optional[UP]: 

100 """Get a single user by email.""" 

101 statement = select(self.user_model).where( 

102 func.lower(self.user_model.email) == func.lower(email) 

103 ) 

104 results = await self.session.execute(statement) 

105 object = results.first() 

106 if object is None: 

107 return None 

108 return object[0] 

109 

110 async def get_by_oauth_account( 

111 self, oauth: str, account_id: str 

112 ) -> Optional[UP]: # noqa 

113 """Get a single user by OAuth account id.""" 

114 if self.oauth_account_model is None: 

115 raise NotImplementedError() 

116 statement = ( 

117 select(self.oauth_account_model) 

118 .where(self.oauth_account_model.oauth_name == oauth) 

119 .where(self.oauth_account_model.account_id == account_id) 

120 .options(selectinload(self.oauth_account_model.user)) # type: ignore # noqa 

121 ) 

122 results = await self.session.execute(statement) 

123 oauth_account = results.first() 

124 if oauth_account: 

125 user = oauth_account[0].user # type: ignore 

126 return user 

127 return None

255async def _create_first_user( 

256 email: str, 

257 password: str, 

258 is_superuser: bool = False, 

259 is_verified: bool = False, 

260 username: Optional[str] = None, 

261) -> None: 

262 """ 

263 Private method to create the first fractal-server user 

264 

265 Create a user with the given default arguments and return a message with 

266 the relevant informations. If the user alredy exists, for example after a 

267 restart, it returns a message to inform that user already exists. 

268 

269 **WARNING**: This function is only meant to create the first user, and then 

270 it catches and ignores `IntegrityError`s (when multiple workers may be 

271 trying to concurrently create the first user). This is not the expected 

272 behavior for regular user creation, which must rather happen via the 

273 /auth/register endpoint. 

274 

275 See [fastapi_users docs](https://fastapi-users.github.io/fastapi-users/ 

276 12.1/cookbook/create-user-programmatically) 

277 

278 Arguments: 

279 email: New user's email 

280 password: New user's password 

281 is_superuser: `True` if the new user is a superuser 

282 is_verified: `True` if the new user is verifie 

283 username: 

284 """ 

285 try: 

286 async with get_async_session_context() as session: 

287 

288 if is_superuser is True: 

289 # If a superuser already exists, exit 

290 stm = select(UserOAuth).where( 

291 UserOAuth.is_superuser == True # noqa: E712 

292 ) 

293 res = await session.execute(stm) 

294 existing_superuser = res.scalars().first() 

295 if existing_superuser is not None: 

296 logger.info( 

297 f"{existing_superuser.email} superuser already exists," 

298 f" skip creation of {email}" 

299 ) 

300 return None 

301 

302 async with get_user_db_context(session) as user_db: 

303 async with get_user_manager_context(user_db) as user_manager: 

304 kwargs = dict( 

305 email=email, 

306 password=password, 

307 is_superuser=is_superuser, 

308 is_verified=is_verified, 

309 ) 

310 if username is not None: 

311 kwargs["username"] = username 

312 user = await user_manager.create(UserCreate(**kwargs)) 

313 logger.info(f"User {user.email} created") 

314 

315 except IntegrityError: 315 ↛ 316line 315 didn't jump to line 316, because the exception caught by line 315 didn't happen

316 logger.warning( 

317 f"Creation of user {email} failed with IntegrityError " 

318 "(likely due to concurrent attempts from different workers)." 

319 ) 

320 

321 except UserAlreadyExists: 

322 logger.warning(f"User {email} already exists")
