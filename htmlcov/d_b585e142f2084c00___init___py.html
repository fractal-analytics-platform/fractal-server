12""" 

13Auth subsystem 

14 

15This module implements the authorisation/authentication subsystem of the 

16Fractal Server. It is based on the 

17[FastAPI Users](https://fastapi-users.github.io/fastapi-users/) 

18library with 

19[support](https://github.com/fastapi-users/fastapi-users-db-sqlmodel) for the 

20SQLModel database adapter. 

21 

22In particular, this module links the appropriate database models, sets up 

23FastAPIUsers with Barer Token and cookie transports and register local routes. 

24Then, for each OAuth client defined in the Fractal Settings configuration, it 

25registers the client and the relative routes. 

26

64class SQLModelUserDatabaseAsync(Generic[UP, ID], BaseUserDatabase[UP, ID]): 

65 """ 

66 This class is from fastapi_users_db_sqlmodel 

67 Original Copyright: 2022 François Voron, released under MIT licence 

68 

69 Database adapter for SQLModel working purely asynchronously. 

70 

71 Args: 

72 user_model: SQLModel model of a DB representation of a user. 

73 session: SQLAlchemy async session. 

74 """ 

75 

76 session: AsyncSession 

77 user_model: Type[UP] 

78 oauth_account_model: Optional[Type[OAuthAccount]] 

79 

80 def __init__( 

81 self, 

82 session: AsyncSession, 

83 user_model: Type[UP], 

84 oauth_account_model: Optional[Type[OAuthAccount]] = None, 

85 ): 

86 self.session = session 

87 self.user_model = user_model 

88 self.oauth_account_model = oauth_account_model 

89 

90 async def get(self, id: ID) -> Optional[UP]: 

91 """Get a single user by id.""" 

92 return await self.session.get(self.user_model, id) 

93 

94 async def get_by_email(self, email: str) -> Optional[UP]: 

95 """Get a single user by email.""" 

96 statement = select(self.user_model).where( 

97 func.lower(self.user_model.email) == func.lower(email) 

98 ) 

99 results = await self.session.execute(statement) 

100 object = results.first() 

101 if object is None: 

102 return None 

103 return object[0] 

104 

105 async def get_by_oauth_account( 

106 self, oauth: str, account_id: str 

107 ) -> Optional[UP]: # noqa 

108 """Get a single user by OAuth account id.""" 

109 if self.oauth_account_model is None: 

110 raise NotImplementedError() 

111 statement = ( 

112 select(self.oauth_account_model) 

113 .where(self.oauth_account_model.oauth_name == oauth) 

114 .where(self.oauth_account_model.account_id == account_id) 

115 .options(selectinload(self.oauth_account_model.user)) # type: ignore # noqa 

116 ) 

117 results = await self.session.execute(statement) 

118 oauth_account = results.first() 

119 if oauth_account: 

120 user = oauth_account[0].user # type: ignore 

121 return user 

122 return None 

123 

124 async def create(self, create_dict: Dict[str, Any]) -> UP: 

125 """Create a user.""" 

126 user = self.user_model(**create_dict) 

127 self.session.add(user) 

128 await self.session.commit() 

129 await self.session.refresh(user) 

130 return user 

131 

132 async def update(self, user: UP, update_dict: Dict[str, Any]) -> UP: 

133 for key, value in update_dict.items(): 

134 setattr(user, key, value) 

135 self.session.add(user) 

136 await self.session.commit() 

137 await self.session.refresh(user) 

138 return user 

139 

140 async def delete(self, user: UP) -> None: 

141 await self.session.delete(user) 

142 await self.session.commit() 

143 

144 async def add_oauth_account( 

145 self, user: UP, create_dict: Dict[str, Any] 

146 ) -> UP: # noqa 

147 if self.oauth_account_model is None: 

148 raise NotImplementedError() 

149 

150 oauth_account = self.oauth_account_model(**create_dict) 

151 user.oauth_accounts.append(oauth_account) # type: ignore 

152 self.session.add(user) 

153 

154 await self.session.commit() 

155 

156 return user 

157 

158 async def update_oauth_account( 

159 self, user: UP, oauth_account: OAP, update_dict: Dict[str, Any] 

160 ) -> UP: 

161 if self.oauth_account_model is None: 

162 raise NotImplementedError() 

163 

164 for key, value in update_dict.items(): 

165 setattr(oauth_account, key, value) 

166 self.session.add(oauth_account) 

167 await self.session.commit() 

168 

169 return user
